### 資料庫設計（統一管理）
```sql
-- 統一的資料結構
CREATE TABLE properties (
  id SERIAL PRIMARY KEY,
  name VARCHAR NOT NULL,
  address TEXT,
  template_id INTEGER,  -- 模板 ID
  settings TEXT,        -- JSON 配置
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE rooms (
  id SERIAL PRIMARY KEY,
  property_id INTEGER REFERENCES properties(id),
  floor INTEGER,
  room_number VARCHAR NOT NULL,
  monthly_rent DECIMAL,
  deposit DECIMAL,
  status VARCHAR,
  tenant_name VARCHAR,
  contract_start DATE,
  contract_end DATE,
  check_in_date DATE,
  check_out_date DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE tenants (
  id SERIAL PRIMARY KEY,
  room_id INTEGER REFERENCES rooms(id),
  name VARCHAR NOT NULL,
  phone VARCHAR,
  nationality VARCHAR,
  contract_start DATE,
  contract_end DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE charges (
  id SERIAL PRIMARY KEY,
  type VARCHAR NOT NULL,  -- 月租收入、洗衣局機等
  room_id INTEGER REFERENCES rooms(id),
  amount DECIMAL NOT NULL,
  status VARCHAR,         -- 已收、等待、逾期
  due_date DATE,
  paid_date DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE expenses (
  id SERIAL PRIMARY KEY,
  type VARCHAR NOT NULL,
  property_id INTEGER REFERENCES properties(id),
  amount DECIMAL NOT NULL,
  due_date DATE,
  paid_date DATE,
  created_at TIMESTAMP DEFAULT NOW()
);
```
